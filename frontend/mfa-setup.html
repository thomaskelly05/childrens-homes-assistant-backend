@router.post("/setup", response_model=MfaSetupResponse) async def mfa_setup(request: Request): user_id, email = _require_session_user(request) try: secret = pyotp.random_base32() except Exception: raise HTTPException( status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, detail="Could not generate MFA secret.", ) try: upsert_user_mfa_secret(user_id=user_id, email=email, totp_secret=secret) except Exception as exc: print("MFA SETUP DB ERROR:", exc) raise HTTPException( status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, detail="Could not save MFA setup.", ) try: label = email or f"user-{user_id}" otp_url = pyotp.TOTP(secret).provisioning_uri( name=label, issuer_name=TOTP_ISSUER, ) qr_code_data_url = _build_qr_data_url(otp_url) except Exception as exc: print("MFA QR ERROR:", exc) raise HTTPException( status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, detail="Could not generate MFA QR code.", ) try: _log_mfa_event( request, user_id=user_id, email=email, event_type="mfa_setup_started", detail="TOTP setup initiated", ) except Exception as exc: print("MFA LOG ERROR:", exc) return MfaSetupResponse( ok=True, secret=secret, otp_auth_url=otp_url, qr_code_data_url=qr_code_data_url, )
